=== 학습목표
* 포인터란
* 32bit 컴퓨터와 64bit 컴퓨터에서 포인터의 크기와 역할에 대해서 알고있다.


////
[source,c++]
----
#include <iostream>

int main()
{
    void* ptr = malloc(16);
    if (ptr == NULL)
        return 1;

    int* ptrInt = (int*)ptr;
    std::cout << "[1] = " << (int)ptr << std::endl;
    std::cout << "[2] = " << (int)ptr + 4 << std::endl;
    std::cout << "[3] = " << (int)ptrInt + 4 << std::endl;
    std::cout << "[4] = " << (int)&ptrInt[1] << std::endl;

    return 0;
}
